 64    def find_all(self, include_inactive: bool = False) -> List[Habit]:
 65        """
 66        Returns all habits from the database.
 67        Daily habits first, then weekly. Within each group, the newest first.
 68
 69        Args:
 70            include_inactive: Whether to include inactive habits
 71
 72        Returns:
 73            List of Habit objects ordered by periodicity (daily first), then by creation date (the newest first)
 74        """
 75        con = self.db or Database.get_connection()
 76        cur = con.cursor()
 77
 78        cur.execute("""
 79            SELECT habit_id, name, periodicity, created_at, updated_at, is_active, description
 80            FROM habits
 81        """)
 82
 83        results = cur.fetchall()
 84        if not self.db:
 85            con.close()
 86
 87        # Functional approach: map, filter, sort
 88        habits = map(Habit.from_tuple, results)
 89
 90        # Filter inactive habits if needed
 91        if not include_inactive:
 92            habits = filter(lambda h: h.is_active, habits)
 93
 94        # Define sorting key function (functional approach)
 95        def get_sort_key(habit: Habit) -> tuple:
 96            """
 97            Returns tuple for sorting:
 98            - First element: periodicity priority (daily=1, weekly=2, other=3)
 99            - Second element: negative timestamp (for descending order)
100            """
101            periodicity_map = {'daily': 1, 'weekly': 2}
102            periodicity_priority = periodicity_map.get(habit.periodicity, 3)
103            creation_time = -habit.created_at.timestamp()  # Negative for descending
104
105            return periodicity_priority, creation_time
106
107        # Sort using a functional key
108        return sorted(habits, key=get_sort_key)
