109    def show_completion_statistics(self, summary):
110        """
111        Display comprehensive completion statistics.
112
113        Args:
114            summary: List of dictionaries with habit statistics
115        """
116
117        self.show_header("üìà [bold cyan]Completion Statistics[/bold cyan]")
118
119        if not summary:
120            self.console.print("  No habits found.", style="dim")
121            return
122
123        table = Table(
124            show_header=True,
125            header_style="bold magenta",
126            box=box.ROUNDED,
127            padding=(0, 2),
128            expand=False
129        )
130
131        table.add_column("", width=3, justify="center")
132        table.add_column("Habit Name", style="cyan bold", min_width=20, justify="center")
133        table.add_column("Type", style="yellow", width=10, justify="center")
134        table.add_column("Total", style="green", width=8, justify="center")
135        table.add_column("Current", style="blue", width=10, justify="center")
136        table.add_column("Longest", style="magenta", width=10, justify="center")
137        table.add_column("Last Done", style="white", width=12, justify="center")
138
139        for item in summary:
140            icon = get_periodicity_icon(item['periodicity'])
141            name = item['name']
142            periodicity = item['periodicity'].capitalize()
143            total = str(item.get('total_completions', 0))
144
145            # Format streaks
146            current_streak = item.get('current_streak', 0)
147            longest_streak = item.get('longest_streak', 0)
148            current_str = f"{current_streak} {'day' if item['periodicity'] == 'daily' else 'week'}{'s' if current_streak != 1 else ''}"
149            longest_str = f"{longest_streak} {'day' if item['periodicity'] == 'daily' else 'week'}{'s' if longest_streak != 1 else ''}"
150
151            # Format last completion
152            last_completion = item.get('last_completion')
153            if last_completion:
154                if isinstance(last_completion, str):
155                    last_completion = datetime.fromisoformat(last_completion)
156                days_ago = (datetime.now() - last_completion).days
157                if days_ago == 0:
158                    last_done = "Today"
159                elif days_ago == 1:
160                    last_done = "Yesterday"
161                else:
162                    last_done = f"{days_ago}d ago"
163            else:
164                last_done = "Never"
165
166            table.add_row(icon, name, periodicity, total, current_str, longest_str, last_done)
167
168        self.console.print(table)
169        self.console.print()
