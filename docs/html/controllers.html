 50    def delete_habit(self):
 51        """Interactive habit deletion with soft/hard delete choice."""
 52        while True:
 53            self.view.show_header("üóëÔ∏è  [bold red]Delete a habit[/bold red]")
 54
 55            habits = self.service.get_all_habits(include_inactive=True)
 56            if not habits:
 57                self.view.show_no_habits_found()
 58                return
 59
 60            # Convert to tuples for display
 61            habit_tuples = [(h.name, h.periodicity, h.is_active) for h in habits]
 62            self.view.show_habits_numbered_list_with_status(habit_tuples)
 63
 64            choice = self.view.get_number_choice(
 65                "\nEnter the number of the habit to delete (or 'q' to quit): "
 66            )
 67
 68            if choice.lower() == 'q':
 69                return
 70
 71            try:
 72                choice_num = int(choice)
 73                if 1 <= choice_num <= len(habits):
 74                    selected_habit = habits[choice_num - 1]
 75                    name = selected_habit.name
 76
 77                    # Ask for a deletion type
 78                    self.view.console.print()
 79                    deletion_type = self.view.get_deletion_type()
 80
 81                    if deletion_type == 'cancel':
 82                        self.view.console.print("‚ùå Deletion cancelled.", style="yellow")
 83                        return
 84
 85                    # Confirm deletion
 86                    if deletion_type == 'soft':
 87                        confirm_message = f"‚ö†Ô∏è  Archive '{name}'? (You can restore it later) (y/n): "
 88                    else:
 89                        confirm_message = f"‚ö†Ô∏è  PERMANENTLY delete '{name}' and all its data? This cannot be undone!  (y/n): "
 90
 91                    confirm = self.view.get_confirmation(confirm_message)
 92
 93                    if confirm.lower() != 'y':
 94                        self.view.console.print("‚ùå Deletion cancelled.", style="yellow")
 95                        return
 96
 97                    # Perform deletion
 98                    soft_delete = (deletion_type == 'soft')
 99                    success, message = self.service.delete_habit(name, soft_delete=soft_delete)
100
101                    if success:
102                        if soft_delete:
103                            self.view.show_habit_archived(name)
104                        else:
105                            self.view.show_habit_deleted(name)
106                    else:
107                        self.view.show_error(message)
108                    return
109                else:
110                    self.view.show_error(
111                        f"Invalid number. Please enter a number between 1 and {len(habits)}."
112                    )
113                    self.view.show_retry_message()
114            except ValueError:
115                self.view.show_error("Invalid input. Please enter a number.")
116                self.view.show_retry_message()
