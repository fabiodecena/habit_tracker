 26    def create_habit(self, name: str, periodicity: str, description: str = "") -> Tuple[bool, str]:
 27        """
 28        Creates a new habit with validation.
 29
 30        Args:
 31            name: Habit name
 32            periodicity: 'daily' or 'weekly'
 33            description: Optional description
 34
 35        Returns:
 36            Tuple of (success:   bool, message: str)
 37        """
 38        # Validation
 39        if not name or not name.strip():
 40            return False, "Habit name cannot be empty"
 41
 42        if periodicity not in Config.DEFAULT_PERIODICITY_OPTIONS:
 43            return False, f"Periodicity must be one of {Config.DEFAULT_PERIODICITY_OPTIONS}"
 44
 45        # Check if the habit already exists
 46        existing = self.repository.find_by_name(name)
 47        if existing:
 48            return False, f"Habit '{name}' already exists"
 49
 50        # Create and save
 51        habit = Habit(name=name.strip(), periodicity=periodicity, description=description.strip())
 52        success = self.repository.save(habit)
 53
 54        if success:
 55            return True, f"Habit '{name}' created successfully"
 56        else:
 57            return False, "Failed to create habit"

 65        new_description: str = ""
 66    ) -> Tuple[bool, str]:
 67        """
 68        Updates an existing habit with validation.
 69
 70        Args:

 27    def check_off_habit(
 28            self,
 29            habit_name: str,
 30            checked_at: datetime = None,
 31            notes: str = ""
 32    ) -> Tuple[bool, str]:
 33        """
 34        Records a habit completion with validation.
 35
 36        Args:
 37            habit_name: Name of the habit
 38            checked_at: When completed (defaults to now)
 39            notes: Optional notes about this completion
 40
 41        Returns:
 42            Tuple of (success:  bool, message: str)
 43        """
 44        # Check if the habit exists
 45        habit = self.habit_repo.find_by_name(habit_name)
 46        if not habit:
 47            return False, f"Habit '{habit_name}' not found"
 48
 49        # Check if habit is active
 50        if not habit.is_active:
 51            return False, f"Habit '{habit_name}' is inactive"
 52
 53        # Default to now if not specified
 54        if checked_at is None:
 55            checked_at = datetime.now()
 56
 57        # Validate date is not in the future
 58        if checked_at > datetime.now():
 59            return False, "Cannot check off a habit in the future"
 60
 61        # Create and save event
 62        event = TrackerEvent(
 63            habit_id=habit.habit_id,
 64            checked_at=checked_at,
 65            notes=notes
 66        )
 67        success = self.tracker_repo.save(event)
 68
 69        if success:
 70            return True, f"Habit '{habit_name}' checked off successfully"
 71        else:
 72            return False, "Failed to check off habit"
 73

 26    def calculate_longest_streak(self, habit_name: str) -> int:
 27        """
 28        Calculates the longest streak for a habit.
 29
 30        Args:
 31            habit_name: Name of the habit
 32
 33        Returns:
 34            Length of the longest streak
 35        """
 36        habit = self.habit_repo.find_by_name(habit_name)
 37        if not habit:
 38            return 0
 39
 40        events = self.tracker_repo.find_by_habit_id(habit.habit_id)  # Use habit_id
 41        if not events:
 42            return 0
 43
 44        history = [event.checked_at for event in events]
 45        periodicity = habit.periodicity
 46
 47        # Normalize dates based on periodicity
 48        if periodicity == 'daily':
 49            normalized_dates = sorted({d.date() for d in history})
 50            step = timedelta(days=1)
 51        else:
 52            # Normalize to the start (Monday) of the ISO week
 53            normalized_dates = sorted({
 54                datetime.fromisocalendar(
 55                    d.isocalendar()[0],
 56                    d.isocalendar()[1],
 57                    1
 58                ).date()
 59                for d in history
 60            })
 61            step = timedelta(weeks=1)
 62
 63        if len(normalized_dates) < 2:
 64            return len(normalized_dates)
 65
 66        longest_streak = 0
 67        current_streak = 1
 68
 69        for i in range(1, len(normalized_dates)):
 70            prev = normalized_dates[i - 1]
 71            curr = normalized_dates[i]
 72
 73            if curr - prev == step:
 74                current_streak += 1
 75            else:
 76                longest_streak = max(longest_streak, current_streak)
 77                current_streak = 1
 78
 79        return max(longest_streak, current_streak)

27        """
28        Calculates the longest streak for a habit.
29
30        Args:
31            habit_name: Name of the habit
32
33        Returns:
34            Length of the longest streak
35        """
36        habit = self.habit_repo.find_by_name(habit_name)
37        if not habit:
38            return 0
39
40        events = self.tracker_repo.find_by_habit_id(habit.habit_id)  # Use habit_id
41        if not events:
42            return 0
43
44        history = [event.checked_at for event in events]
45        periodicity = habit.periodicity
46
47        # Normalize dates based on periodicity
48        if periodicity == 'daily':
49            normalized_dates = sorted({d.date() for d in history})
50            step = timedelta(days=1)
51        else:
52            # Normalize to the start (Monday) of the ISO week
53            normalized_dates = sorted({
54                datetime.fromisocalendar(
55                    d.isocalendar()[0],
56                    d.isocalendar()[1],
57                    1
58                ).date()
59                for d in history
60            })
61            step = timedelta(weeks=1)
62
63        if len(normalized_dates) < 2:
64            return len(normalized_dates)
65
66        longest_streak = 0
67        current_streak = 1
68
69        for i in range(1, len(normalized_dates)):
70            prev = normalized_dates[i - 1]
71            curr = normalized_dates[i]
72
73            if curr - prev == step:
74                current_streak += 1
75            else:
76                longest_streak = max(longest_streak, current_streak)
77                current_streak = 1
78
79        return max(longest_streak, current_streak)
